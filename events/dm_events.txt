	id = demilitarization.1
	hide_window = yes
	is_triggered_only = yes

			limit = {
				is_guaranteeing = event_target:loser
			}

				id = demilitarization.1
				days = 30
			}
		}
		else_if = {
			limit = {
				has_modifier = dm_demilitarizer
			}
			remove_modifier = dm_demilitarizer
			# TODO notify
		}
	}
}
