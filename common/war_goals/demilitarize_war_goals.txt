
wg_demilitarize = {
	casus_belli = cb_demilitarize

	hide_if_no_cb = yes

	surrender_acceptance = -75
	allowed_peace_offers = { 
		surrender
		status_quo
		demand_surrender
	}

	potential = {
		NOR = {
			has_valid_civic = civic_inwards_perfection
			has_authority = auth_corporate
		}
		has_total_war_cb = no
		from = {
			NOT = { has_modifier = dm_demilitarized }
		}
	}
	possible = {
		always = yes
	}
	on_accept = {
        custom_tooltip = wg_demilitarize_accept
        hidden_effect = {
			add_resource = { influence = 150 }
			add_modifier = {
				modifier = dm_demilitarizer
				days = 3600
			}
            from = {
				if = {
					limit = { is_militarist = yes }
					add_modifier = { modifier = dm_demilitarized_mil days = 3600 }
				}
				else = {
					add_modifier = { modifier = dm_demilitarized days = 3600 }
				}
				set_policy = {
					policy = war_philosophy
					option = no_wars
					cooldown = yes
				}
			}
			guarantee_country = {
				target = from
				free_guarantee_days = 3600
			}
			# set_closed_borders = {
			# 	who = from
			# 	status = no
			# }
		}
	}
 
	ai_weight = {
		weight = 5
		modifier = {
			factor = 2
			from = { has_ethic = ethic_militarist }
		}
		modifier = {
			factor = 5
			from = { has_ethic = ethic_fanatic_militarist }
		}
	}
}
